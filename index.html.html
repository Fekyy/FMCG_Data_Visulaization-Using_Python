import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
%matplotlib inline
from sqlalchemy import create_engine


db_connection_str = 'mysql+mysqlconnector://root:123456@localhost/supply_chain_db'
db_conn = create_engine(db_connection_str)


query = """
SELECT 
    f.order_id, 
    f.order_date, 
    f.delivery_status, 
    f.late_delivery_risk, 
    f.shipping_mode, 
    f.sales, 
    f.order_profit,           
    f.product_id, 
    c.customer_city, 
    c.customer_segment, 
    p.product_category,
    p.product_name            
FROM fact_orders f
JOIN dim_customers c ON c.customer_id = f.customer_id
JOIN dim_products p ON p.product_id = f.product_id
"""

product_revenue = df_master.groupby('product_id')['sales'].sum().reset_index()
product_revenue = product_revenue.sort_values('sales', ascending=False)


product_revenue['cumulative_sales'] = product_revenue['sales'].cumsum()
product_revenue['total_sales'] = product_revenue['sales'].sum()
product_revenue['cumulative_percent'] = (product_revenue['cumulative_sales'] / product_revenue['total_sales']) * 100


def assign_class(percent):
    if percent <= 80: return 'A'
    elif percent <= 95: return 'B'
    else: return 'C'

product_revenue['abc_class'] = product_revenue['cumulative_percent'].apply(assign_class)


plt.figure(figsize=(8, 5))
sns.countplot(x='abc_class', data=product_revenue, palette='viridis', hue='abc_class', order=['A', 'B', 'C'])

plt.title('ABC Analysis: Product Count by Class', fontsize=16, fontweight='bold')
plt.xlabel('Class (A = Top 80% Revenue)', fontsize=12)
plt.ylabel('Number of Unique Products', fontsize=12)
plt.show()

def assign_profit_status(profit):
    if profit > 0:
        return "Profit"
    elif profit < 0:
        return "Loss"
    else:
        return "Break Even"

df_master['profit_status'] = df_master['order_profit'].apply(assign_profit_status)


status_counts = df_master['profit_status'].value_counts()


plt.figure(figsize=(7, 7))
plt.pie(
    status_counts, 
    labels=status_counts.index, 
    autopct='%1.1f%%', 
    colors=['#66b3ff', '#ff9999', '#99ff99'], # Blue, Red, Green
    startangle=90
)

plt.title('Order Profitability', fontsize=16, fontweight='bold')
plt.show()

query = """select product_name , count(order_profit) as Num_of_products, sum(order_profit) as total_profit
from supply_chain_db.fact_orders f
join supply_chain_db.dim_products p on p.product_id = f.product_id
where order_profit < 0
group by product_name
order by total_profit
limit 10;
"""
